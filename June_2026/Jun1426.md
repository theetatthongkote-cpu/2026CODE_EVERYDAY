from storage import check_if_question_exists


def test_check_if_question_exists():
    # Test case 1: Valid question data
    question_data_valid = {
        'Question Code': 'Q001',
        'Question Type': 'Math',
        'Question Difficulty': 'Medium',
        'Question': 'What is 2 + 2?'
    }
    assert check_if_question_exists(
        question_data_valid) == False, "Test case 1 failed"

    # Test case 2: Empty fields
    question_data_empty_fields = {
        'Question Code': '',
        'Question Type': '',
        'Question Difficulty': '',
        'Question': ''
    }
    assert check_if_question_exists(
        question_data_empty_fields) == True, "Test case 2 failed"

    # Test case 3: Duplicate question code
    question_data_duplicate_code = {
        'Question Code': 'Q001',  # Assuming Q001 already exists in the JSON file
        'Question Type': 'Science',
        'Question Difficulty': 'Hard',
        'Question': 'What is the chemical symbol for water?'
    }
    assert check_if_question_exists(
        question_data_duplicate_code) == True, "Test case 3 failed"

    print("All test cases passed!")
